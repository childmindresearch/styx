236        buf = []
237
238        # Sort arguments so default arguments come last
239        func.args.sort(key=lambda a: a.default is not None)
240
241        # Function signature
242        buf.append(f"def {func.name}(")
243
244        # Add arguments
245        for arg in func.args:
246            buf.extend(indent([f"{self.generate_arg_declaration(arg)},"]))
247        buf.append(f") -> {func.return_type}:")
248
249        arg_docstr_buf = []
250        for arg in func.args:
251            if arg.name == "self":
252                continue
253            arg_docstr = linebreak_paragraph(
254                f"{arg.name}: {escape_backslash(arg.docstring) if arg.docstring else ''}",
255                width=80 - (4 * 3) - 1,
256                first_line_width=80 - (4 * 2) - 1,
257            )
258            arg_docstr = ensure_endswith("\\\n".join(arg_docstr), ".").split("\n")
259            arg_docstr_buf.append(arg_docstr[0])
260            arg_docstr_buf.extend(indent(arg_docstr[1:]))
261
262        # Add docstring (Google style)
263
264        if func.docstring_body:
265            docstring_linebroken = linebreak_paragraph(escape_backslash(func.docstring_body), width=80 - 4)
266        else:
267            docstring_linebroken = [""]
268
269        buf.extend(
270            indent([
271                '"""',
272                *docstring_linebroken,
273                "",
274                "Args:",
275                *indent(arg_docstr_buf),
276                *(["Returns:", *indent([f"{escape_backslash(func.return_descr)}"])] if func.return_descr else []),
277                '"""',
278            ])
279        )
280
281        # Add function body
282        if func.body:
283            buf.extend(indent(func.body))
284        else:
285            buf.extend(indent(["pass"]))
286        return buf

394        def _val() -> MStr:
395            if not param.list_:
396                if isinstance(param.body, ir.Param.String):
397                    return MStr(symbol, False)
398                if isinstance(param.body, (ir.Param.Int, ir.Param.Float)):
399                    return MStr(f"str({symbol})", False)
400                if isinstance(param.body, ir.Param.Bool):
401                    as_list = (len(param.body.value_true) > 1) or (len(param.body.value_false) > 1)
402                    if as_list:
403                        value_true: str | list[str] | None = param.body.value_true
404                        value_false: str | list[str] | None = param.body.value_false
405                    else:
406                        value_true = param.body.value_true[0] if len(param.body.value_true) > 0 else None
407                        value_false = param.body.value_false[0] if len(param.body.value_false) > 0 else None
408                    if len(param.body.value_true) > 0:
409                        if len(param.body.value_false) > 0:
410                            return MStr(
411                                f"({self.expr_literal(value_true)} if {symbol} else {self.expr_literal(value_true)})",
412                                as_list,
413                            )
414                        return MStr(self.expr_literal(value_true), as_list)
415                    assert len(param.body.value_false) > 0
416                    return MStr(self.expr_literal(value_false), as_list)
417                if isinstance(param.body, ir.Param.File):
418                    extra_args = ""
419                    if param.body.resolve_parent:
420                        extra_args += ", resolve_parent=True"
421                    if param.body.mutable:
422                        extra_args += ", mutable=True"
