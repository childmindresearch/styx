190        """Iterate through all child-params recursively."""
191        if not skip_self:
192            yield self
193        if isinstance(self.body, Param.Struct):

195                yield from e.iter_params_recursively(False)

207    def __init__(
208        self,
209        base: Base,
210        body: T,
211        list_: Optional[List] = None,
212        nullable: bool = False,
213        choices: Optional[list[Union[str, int, float]]] = None,
214        default_value: Union[
215            bool, str, int, float, list[bool], list[str], list[int], list[float], type[SetToNone], None
216        ] = None,
217    ) -> None:
218        """Initialize a Param instance.
219
220        Args:
221            base: Base parameter information.
222            body: The body of the parameter. Determines its base type.
223            list_: List information if the parameter is a list.
224            nullable: Whether the parameter can be null.
225            choices: List of possible choices for the parameter.
226            default_value: Default value for the parameter.
227
228        Raises:
229            TypeError: If any of the input types are incorrect.
230            ValueError: If there are constraint violations.
231        """
232        self.base = base
233        self.body: T = body
234        self.list_ = list_
235        self.nullable = nullable
236        self.choices = choices
237        self.default_value = default_value
238
239        # Runtime type checking
240        self._check_base()
241        self._check_body_type()
242        self._check_list()
243        self._check_nullable()
244        self._check_choices()
245        self._check_default_value()
246        self._check_constraints()
247
248    def _check_base(self) -> None:
249        """Check if base is an instance of Param.Base."""
250        if not isinstance(self.base, Param.Base):
251            raise TypeError("base must be an instance of Param.Base")
252
253    def _check_body_type(self) -> None:
254        """Check if body is an instance of one of the allowed types."""
255        if not isinstance(
256            self.body, (Param.Bool, Param.Int, Param.Float, Param.String, Param.File, Param.Struct, Param.StructUnion)
257        ):
258            raise TypeError(
259                "body must be an instance of "
260                "Param.Bool, Param.Int, Param.Float, Param.String, Param.File, Param.Struct or Param.StructUnion"
261            )
262
263    def _check_list(self) -> None:
264        """Check if list_ is None or an instance of Param.List."""
265        if self.list_ is not None and not isinstance(self.list_, Param.List):
266            raise TypeError("list_ must be None or an instance of Param.List")
267
268    def _check_nullable(self) -> None:
269        """Check if nullable is a boolean."""
270        if not isinstance(self.nullable, bool):
271            raise TypeError("nullable must be a boolean")
272
273    def _check_choices(self) -> None:
274        """Check if choices is None or a list of the correct type."""
275        if self.choices is not None:
276            if not isinstance(self.choices, list):
277                raise TypeError("choices must be None or a list")
278            expected_type = self._get_expected_type()
279            if expected_type is not None and not all(isinstance(choice, expected_type) for choice in self.choices):
280                raise TypeError(f"All choices must be of type {' or '.join([e.__name__ for e in expected_type])}")
281
282    def _check_default_value(self) -> None:
283        """Check if default_value is of the correct type."""
284        if self.default_value is None:
285            return
286        if self.default_value is Param.SetToNone:
287            if not self.nullable:
288                raise ValueError("default_value cannot be SetToNone when nullable is False")
289            return
290
291        expected_type = self._get_expected_type()
292        if expected_type is None:
293            raise TypeError("default_value must be a None for this type")
294        if self.list_:
295            if not isinstance(self.default_value, list):
296                raise TypeError("default_value must be a list when list_ is provided")
297            if not all(isinstance(item, expected_type) for item in self.default_value):
298                raise TypeError(
299                    f"All items in default_value must be of type {' or '.join([e.__name__ for e in expected_type])}"
300                )
301        else:
302            if not isinstance(self.default_value, expected_type):
303                raise TypeError(f"default_value must be of type {' or '.join([e.__name__ for e in expected_type])}")
304
305    def _check_constraints(self) -> None:
306        """Check if all constraints are satisfied."""
307        if isinstance(self.body, (Param.Int, Param.Float)):
308            if self.body.min_value is not None and self.body.max_value is not None:
309                if self.body.min_value > self.body.max_value:
310                    raise ValueError("min_value cannot be greater than max_value")
311
312            if (
313                self.default_value is not None
314                and self.default_value is not Param.SetToNone
315                and not isinstance(self.default_value, (list, Param.SetToNone))
316            ):
317                assert isinstance(self.default_value, (int, float))
318                if self.body.min_value is not None and self.default_value < self.body.min_value:
319                    raise ValueError(f"default_value cannot be less than min_value ({self.body.min_value})")
320                if self.body.max_value is not None and self.default_value > self.body.max_value:
321                    raise ValueError(f"default_value cannot be greater than max_value ({self.body.max_value})")
322
323        if self.list_:
324            if self.list_.count_min is not None and self.list_.count_max is not None:
325                if self.list_.count_min > self.list_.count_max:
326                    raise ValueError("count_min cannot be greater than count_max")
327
328            if isinstance(self.default_value, list):
329                if self.list_.count_min is not None and len(self.default_value) < self.list_.count_min:
330                    raise ValueError(
331                        f"default_value list length cannot be less than count_min ({self.list_.count_min})"
332                    )
333                if self.list_.count_max is not None and len(self.default_value) > self.list_.count_max:
334                    raise ValueError(
335                        f"default_value list length cannot be greater than count_max ({self.list_.count_max})"
336                    )
337
338    def _get_expected_type(self) -> tuple[type, ...] | None:
339        """Get the expected type based on the body type."""
340        if isinstance(self.body, Param.Bool):
341            return (bool,)
342        elif isinstance(self.body, Param.Int):
343            return (int,)
344        elif isinstance(self.body, Param.Float):
345            return float, int
346        elif isinstance(self.body, Param.String):
347            return (str,)
348        elif isinstance(self.body, (Param.File, Param.Struct, Param.StructUnion)):
349            return None
350        else:
351            raise TypeError("Unknown body type")
352
353    def __repr__(self) -> str:
354        """Return a string representation of the Param instance."""
355        parts = [
356            f"Param(id={self.base.id_!r}, name={self.base.name!r}",
357            f"type={self.body}",
358        ]
359
360        if self.list_:
361            parts.append(f"list={self.list_!r}")
362        if self.nullable:
363            parts.append("nullable=True")
364        if self.choices:
365            parts.append(f"choices={self.choices!r}")
366        if self.default_value is not None:
367            parts.append(f"default={self.default_value!r}")
368
369        return ", ".join(parts) + ")"

249    def _check_base(self) -> None:
250        """Check if base is an instance of Param.Base."""
251        if not isinstance(self.base, Param.Base):
252            raise TypeError("base must be an instance of Param.Base")
253
254    def _check_body_type(self) -> None:
255        """Check if body is an instance of one of the allowed types."""
256        if not isinstance(
257            self.body, (Param.Bool, Param.Int, Param.Float, Param.String, Param.File, Param.Struct, Param.StructUnion)
258        ):
259            raise TypeError(

261                "Param.Bool, Param.Int, Param.Float, Param.String, Param.File, Param.Struct or Param.StructUnion"
262            )

265        """Check if list_ is None or an instance of Param.List."""
266        if self.list_ is not None and not isinstance(self.list_, Param.List):
267            raise TypeError("list_ must be None or an instance of Param.List")
268
269    def _check_nullable(self) -> None:
270        """Check if nullable is a boolean."""
271        if not isinstance(self.nullable, bool):
272            raise TypeError("nullable must be a boolean")
273
274    def _check_choices(self) -> None:
275        """Check if choices is None or a list of the correct type."""
276        if self.choices is not None:
277            if not isinstance(self.choices, list):
278                raise TypeError("choices must be None or a list")
279            expected_type = self._get_expected_type()
280            if expected_type is not None and not all(isinstance(choice, expected_type) for choice in self.choices):
281                raise TypeError(f"All choices must be of type {' or '.join([e.__name__ for e in expected_type])}")
282
283    def _check_default_value(self) -> None:
284        """Check if default_value is of the correct type."""
285        if self.default_value is None:
286            return
287        if self.default_value is Param.SetToNone:
288            if not self.nullable:
289                raise ValueError("default_value cannot be SetToNone when nullable is False")
290            return
291
292        expected_type = self._get_expected_type()
293        if expected_type is None:
294            raise TypeError("default_value must be a None for this type")
295        if self.list_:
296            if not isinstance(self.default_value, list):
297                raise TypeError("default_value must be a list when list_ is provided")
298            if not all(isinstance(item, expected_type) for item in self.default_value):
299                raise TypeError(
300                    f"All items in default_value must be of type {' or '.join([e.__name__ for e in expected_type])}"
301                )
302        else:
303            if not isinstance(self.default_value, expected_type):
304                raise TypeError(f"default_value must be of type {' or '.join([e.__name__ for e in expected_type])}")
305
306    def _check_constraints(self) -> None:
307        """Check if all constraints are satisfied."""
308        if isinstance(self.body, (Param.Int, Param.Float)):
309            if self.body.min_value is not None and self.body.max_value is not None:
310                if self.body.min_value > self.body.max_value:
311                    raise ValueError("min_value cannot be greater than max_value")
312
313            if (
314                self.default_value is not None
315                and self.default_value is not Param.SetToNone
316                and not isinstance(self.default_value, (list, Param.SetToNone))
317            ):
318                assert isinstance(self.default_value, (int, float))
319                if self.body.min_value is not None and self.default_value < self.body.min_value:
320                    raise ValueError(f"default_value cannot be less than min_value ({self.body.min_value})")
321                if self.body.max_value is not None and self.default_value > self.body.max_value:
322                    raise ValueError(f"default_value cannot be greater than max_value ({self.body.max_value})")
323
324        if self.list_:
325            if self.list_.count_min is not None and self.list_.count_max is not None:
326                if self.list_.count_min > self.list_.count_max:
327                    raise ValueError("count_min cannot be greater than count_max")
328
329            if isinstance(self.default_value, list):
330                if self.list_.count_min is not None and len(self.default_value) < self.list_.count_min:
331                    raise ValueError(
332                        f"default_value list length cannot be less than count_min ({self.list_.count_min})"
333                    )
334                if self.list_.count_max is not None and len(self.default_value) > self.list_.count_max:
335                    raise ValueError(
336                        f"default_value list length cannot be greater than count_max ({self.list_.count_max})"
337                    )
338
339    def _get_expected_type(self) -> tuple[type, ...] | None:
340        """Get the expected type based on the body type."""
341        if isinstance(self.body, Param.Bool):
342            return (bool,)
343        elif isinstance(self.body, Param.Int):
344            return (int,)
345        elif isinstance(self.body, Param.Float):
346            return float, int
347        elif isinstance(self.body, Param.String):
348            return (str,)
349        elif isinstance(self.body, (Param.File, Param.Struct, Param.StructUnion)):
350            return None
351        else:
352            raise TypeError("Unknown body type")
